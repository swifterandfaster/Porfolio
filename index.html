using UnityEngine;
using UnityEngine.InputSystem;

[RequireComponent(typeof(CharacterController))]
public class PlayerMovement : MonoBehaviour
{
    [Header("Movement Settings")]
    [SerializeField] private float moveSpeed = 5f;
    [SerializeField] private float sprintMultiplier = 1.6f;
    [SerializeField] private float jumpHeight = 2f;
    [SerializeField] private float gravity = -9.81f;

    private CharacterController _controller;
    private Vector3 _velocity;
    private PlayerInputActions _input;
    private bool _isSprinting;

    private void Awake()
    {
        _controller = GetComponent<CharacterController>();
        _input = new PlayerInputActions();
    }

    private void OnEnable() => _input.Enable();
    private void OnDisable() => _input.Disable();

    private void Update()
    {
        Vector2 moveInput = _input.Player.Move.ReadValue<Vector2>();
        _isSprinting = _input.Player.Sprint.IsPressed();

        float currentSpeed = moveSpeed * (_isSprinting ? sprintMultiplier : 1f);
        Vector3 move = transform.right * moveInput.x + transform.forward * moveInput.y;
        _controller.Move(move * currentSpeed * Time.deltaTime);

        if (_controller.isGrounded && _velocity.y < 0f)
            _velocity.y = -2f;

        if (_controller.isGrounded && _input.Player.Jump.WasPressedThisFrame())
            _velocity.y = Mathf.Sqrt(jumpHeight * -2f * gravity);

        _velocity.y += gravity * Time.deltaTime;
        _controller.Move(_velocity * Time.deltaTime);
    }
}

[CreateAssetMenu(fileName = "Inventory", menuName = "Game/Inventory")]
public class InventorySystem : ScriptableObject
{
    [SerializeField] private int maxSlots = 24;
    private List<ItemStack> _slots = new();

    public UnityEvent<ItemStack> onItemAdded;
    public UnityEvent<ItemStack> onItemRemoved;
    public UnityEvent onInventoryFull;

    public bool AddItem(ItemData item, int quantity = 1)
    {
        ItemStack existing = _slots.Find(s => s.data == item && s.count < item.maxStack);
        if (existing != null)
        {
            existing.count += quantity;
            onItemAdded?.Invoke(existing);
            return true;
        }

        if (_slots.Count >= maxSlots)
        {
            onInventoryFull?.Invoke();
            return false;
        }

        ItemStack newStack = new() { data = item, count = quantity };
        _slots.Add(newStack);
        onItemAdded?.Invoke(newStack);
        return true;
    }

    public bool RemoveItem(ItemData item, int quantity = 1)
    {
        ItemStack stack = _slots.Find(s => s.data == item);
        if (stack == null || stack.count < quantity) return false;

        stack.count -= quantity;
        if (stack.count <= 0) _slots.Remove(stack);
        onItemRemoved?.Invoke(stack);
        return true;
    }

    public IReadOnlyList<ItemStack> GetSlots() => _slots.AsReadOnly();
    public int GetItemCount(ItemData item) => _slots.Find(s => s.data == item)?.count ?? 0;
    public bool HasItem(ItemData item, int qty = 1) => GetItemCount(item) >= qty;
    public void Clear() { _slots.Clear(); }
}
using UnityEngine;
using UnityEngine.SceneManagement;
using System;

public class GameManager : MonoBehaviour
{
    public static GameManager Instance { get; private set; }

    public static event Action<GameState> OnStateChanged;

    private GameState _currentState = GameState.MainMenu;
    public GameState CurrentState => _currentState;

    private void Awake()
    {
        if (Instance != null && Instance != this)
        {
            Destroy(gameObject);
            return;
        }
        Instance = this;
        DontDestroyOnLoad(gameObject);
    }

    public void SetState(GameState newState)
    {
        if (_currentState == newState) return;
        _currentState = newState;
        OnStateChanged?.Invoke(_currentState);
        HandleStateTransition(_currentState);
    }

    private void HandleStateTransition(GameState state)
    {
        switch (state)
        {
            case GameState.Playing:
                Time.timeScale = 1f;
                Cursor.lockState = CursorLockMode.Locked;
                break;
            case GameState.Paused:
                Time.timeScale = 0f;
                Cursor.lockState = CursorLockMode.None;
                break;
            case GameState.GameOver:
                Time.timeScale = 0f;
                Cursor.lockState = CursorLockMode.None;
                break;
        }
    }

    public void LoadScene(string sceneName) =>
        SceneManager.LoadSceneAsync(sceneName);

    public void QuitGame() => Application.Quit();
}

public enum GameState { MainMenu, Playing, Paused, GameOver, Loading }
using UnityEngine;
using UnityEngine.XR.Interaction.Toolkit;
using UnityEngine.Events;

[RequireComponent(typeof(XRGrabInteractable))]
public class VRGrabTool : MonoBehaviour
{
    [Header("Tool Settings")]
    [SerializeField] private float toolRange = 0.3f;
    [SerializeField] private LayerMask interactableLayer;
    [SerializeField] private HapticImpulseProvider haptics;

    public UnityEvent<Collider> onToolActivated;
    public UnityEvent onToolDeactivated;

    private XRGrabInteractable _grab;
    private bool _isHeld;

    private void Awake()
    {
        _grab = GetComponent<XRGrabInteractable>();
        _grab.selectEntered.AddListener(_ => _isHeld = true);
        _grab.selectExited.AddListener(_ => _isHeld = false);
        _grab.activated.AddListener(OnActivate);
        _grab.deactivated.AddListener(_ => onToolDeactivated?.Invoke());
    }

    private void OnActivate(ActivateEventArgs args)
    {
        if (!_isHeld) return;

        Collider[] hits = Physics.OverlapSphere(transform.position, toolRange, interactableLayer);
        foreach (var hit in hits)
        {
            onToolActivated?.Invoke(hit);
        }

        haptics?.SendHapticImpulse(0.4f, 0.1f);
    }
}

function createModal({ title, body, onConfirm }) {
  const overlay = document.createElement('div');
  overlay.className = 'modal-overlay';

  const box = document.createElement('div');
  box.className = 'modal-box';
  box.innerHTML = `
    <h2 class="modal-title">${title}</h2>
    <p class="modal-body">${body}</p>
    <div class="modal-actions">
      <button class="btn-cancel">Cancel</button>
      <button class="btn-confirm">Confirm</button>
    </div>
  `;

  overlay.appendChild(box);
  document.body.appendChild(overlay);

  const close = () => overlay.remove();

  box.querySelector('.btn-cancel').addEventListener('click', close);
  box.querySelector('.btn-confirm').addEventListener('click', () => {
    if (typeof onConfirm === 'function') onConfirm();
    close();
  });

  overlay.addEventListener('click', (e) => {
    if (e.target === overlay) close();
  });

  document.addEventListener('keydown', (e) => {
    if (e.key === 'Escape') close();
  }, { once: true });

  return { overlay, close };
}
